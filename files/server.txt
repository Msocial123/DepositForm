apiVersion: v1                  #Use services.yml to expose and access the application
kind: Service
metadata:
  name: deposit-service
spec:
  selector:
    app: deposit
  ports:
    - protocol: TCP
      port: 3000          # Port exposed by the service
      targetPort: 3000    # Port on the container
      nodePort: 32000     # Port on the Minikube node (optional: you can omit this and let Kubernetes assign it automatically)
  type: NodePort         # Use NodePort to expose the service externally
---
